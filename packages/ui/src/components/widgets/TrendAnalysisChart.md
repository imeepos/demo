  type: 'positive' | 'negative' | 'neutral' | 'crisis';
  impact: 'high' | 'medium' | 'low';

  timestamp: Date;
  value: number;
  expectedValue: number;
  deviation: number;
  severity: 'high' | 'medium' | 'low';
}

  name: string;
